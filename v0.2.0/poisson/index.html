Poisson Solvers

using Distributions
using Plots
using PoissonSolvers
using Random
using StatsBase

np = 10000
xp = zeros(np)
wp = ones(np) ./ np
rand!(MersenneTwister(0), Normal(0.5, 0.1), xp)

x = LinRange(0, 1, 100)
y = fit(Histogram, xp, x).weights ./ length(x)
x = x[1:end-1] .+ (x[2] - x[1]) / 2

plot(x, y; xlims = (0,1), xlabel = "x", ylabel = "n", legend = :none)
