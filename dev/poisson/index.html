Poisson Solvers

Solve

\[- \Delta \phi(x) = f(x)\]

using Plots
using PoissonSolvers

sol(x) = sin(π*x) * (1-x^2)
der(x) = π * cos(π*x) * (1-x^2) - 2 * x * sin(π*x)
rhs(x) = 4π * x * cos(π*x) + 2 * sin(π*x) + π^2 * (1-x^2) * sin(π*x)

domain = (-1., +1.)

x = LinRange(domain[begin], domain[end], 100)
